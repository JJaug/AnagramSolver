CREATE TABLE SearchLog (
    UserIp nvarchar (255) NOT NULL,
    Word nvarchar(255) NOT NULL,
	Anagrams int,
	SearchTime nvarchar(255),
	CreatedAt datetime2(7) PRIMARY KEY
);
